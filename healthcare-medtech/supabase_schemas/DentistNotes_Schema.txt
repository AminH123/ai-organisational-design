create table public."Patient Record Data" (
  id bigint generated by default as identity not null,
  created_at timestamp with time zone not null default now(),
  first_name character varying null,
  last_name character varying null,
  date_of_birth timestamp without time zone null default now(),
  identification_num character varying null,
  address_primary character varying null,
  address_secondary character varying null,
  phone character varying null,
  email_primary character varying null,
  constraint Patient Record Data_pkey primary key (id),
  constraint Patient Record Data_id_key unique (id)
) TABLESPACE pg_default;
